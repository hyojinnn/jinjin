# 5. LoopStatement



반복문

1. for문

```jsx
for(let i = 0; i<10; i++) {
    console.log("i : " + i);
}

let j = 0;
for(; j<3; j++) {
    console.log(j);
}

console.log("for문 밖에서 j 호출 : " + j);

/* for문 구성요소 생략하기 */
/* 초기값 for문 밖에 선언 */
let k = 0;
for(; k<3; k++) {
    console.log("초기값 외부에 선언 : " + k);
}

/* 증감[감소] 연산자 위치 변경 */
for(; k<2; ){
    console.log("k++ : " + k++);
}

let m = 0;
/* 조건절을 제거하면 무한반복문 */
for(;;) {
    if(m == 2) {
        console.log("반복종료");
        break;
    }
    m++;
}

for(let i = 0; i <10; i++) {
    if(i %2 == 0) {
        continue;
    }

    console.log("i : " + i);
}
```

2\. while문

```jsx
/*
    while(조건식) {
        실행할문장;
    }
*/

let count = 5;

while(count > 0) {
    console.log("count : " + count);
    count--;
}
```
