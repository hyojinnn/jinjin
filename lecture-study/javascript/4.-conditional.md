# 4. Conditional



조건문

1. if문

```jsx
let number = 1;
if(number > 0) {
    console.log("number는 0보다 큰 수 입니다.");
}

/* 조건식 안에 0, "", null, undefined, NaN -> false로 인지 */
if(0){
    console.log("실행");
}

if(""){
    console.log("실행");
}
```

2\. if - else 절

```jsx
let age = 20;
if(age > 18) {
    console.log("성인입니다.");
} else {
    console.log("성인이 아닙니다.");
}
```

3\. if - else - if 절

```jsx
let checkAge = 10;

if(checkAge > 18) {
    console.log("성인입니다.");
} else if(checkAge > 13) {
    console.log("청소년입니다.");
} else {
    console.log("어린이입니다.");
}
```

4\. 삼항연산자

```jsx
let result = (age > 18 )? "성인입니다" : "성인이 아닙니다";
console.log("result : " + result);
```

5\. 논리연산자

```jsx
/* || (OR)  */
console.log("true || true : " + (true || true));
console.log("true || false : " + (true || false));
console.log("false || true : " + (false || true));
console.log("false || false : " + (false || false));

let lunchTime = 13;

if(lunchTime > 12 || lunchTime < 14) {
    console.log("점심시간입니다.");
}

if(lunchTime < 12 || lunchTime > 14 || true) {
    console.log("점심시간입니다.");
}

/* &&(AND) */
console.log("true && true : " + (true && true));
console.log("true && false : " + (true && false));
console.log("false && true : " + (false && true));
console.log("false && false : " + (false && false));

let breakTime = 10;
let minute = 50;

if(breakTime == 10 && minute == 50) {
    console.log("첫번째 쉬는 쉬간 입니다.");
}
```

6\. !(NOT) 부정연산자

```jsx
console.log("!false : " + !false);
```

7\. switch

```jsx
/* switch(값) {
      case 비교값 : 실행할문장; break;
      case 비교값 : 실행할문장; break;
      case 비교값 : 실행할문장; break;
      default : 실행할문장;
}
*/
let menuNumber = 1;
switch(menuNumber){
    case 1 : console.log("1번 메뉴를 선택하셨습니다.");
        break;
    case 2 : console.log("2번 메뉴를 선택하셨습니다.");
        break;
    default : console.log("입력하신 메뉴 번호가 없습니다.");
        break;
}

let quarterNumber = 9;

switch(quarterNumber) {
    case 1:
    case 2:
    case 3: console.log("1분기입니다.");
        break;
    case 4:
    case 5:
    case 6: console.log("2분기입니다.");
        break;
    case 7:
    case 8:
    case 9: console.log("3분기입니다.");
        break;
    case 10:
    case 11:
    case 12: console.log("4분기입니다.");
        break;
    default : console.log("1-12 사이의 값이 아닙니다.");
        break;
}
```
